Un jeu de serpent avec intrrface graphoque simple gestion du score et des collisions.

import pygame
import time
import random
pygame.init()
white = (255, 255, 255)
yellow = (255, 255, 102)
black = (0, 0, 0)
red = (213, 50, 80)
green = (0, 255, 0)
blue = (50, 153, 213)
width, height = 600, 400
display = pygame.display.set_mode((width, height))
pygame.display.set_caption('Snake Game')
snake_block = 10
snake_speed = 5
clock = pygame.time.Clock()
def our_snake(snake_block, snake_list):
    for x in snake_list:
        pygame.draw.rect(display, black, [x[0], x[1], snake_block, snake_block])
def game_loop():
    game_over = False
    game_close = False
    x1 = width / 2
    y1 = height / 2
    x1_change = 0
    y1_change = 0
    snake_list = []
    length_of_snake = 1
    foodx = round(random.randrange(0, width - snake_block) / 10.0) * 10.0
    foody = round(random.randrange(0, height - snake_block) / 10.0) * 10.0
    while not game_over:
        while game_close == True:
            display.fill(blue)
            font_style = pygame.font.SysFont("bahnschrift", 25)
            mesg = font_style.render("You Lost! Press C-Play Again or Q-Quit", True, red)
            display.blit(mesg, [width / 6, height / 3])
            pygame.display.update()
            for event in pygame.event.get():
                if event.type == pygame.KEYDOWN:
                    if event.key == pygame.K_q:
                        game_over = True
                        game_close = False
                    if event.key == pygame.K_c:
                        game_loop()
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                game_over = True
            if event.type == pygame.KEYDOWN:
                if event.key == pygame.K_LEFT:
                    x1_change = -snake_block
                    y1_change = 0
                elif event.key == pygame.K_RIGHT:
                    x1_change = snake_block
                    y1_change = 0
                elif event.key == pygame.K_UP:
                    y1_change = -snake_block
                    x1_change = 0
                elif event.key == pygame.K_DOWN:
                    y1_change = snake_block
                    x1_change = 0
        if x1 >= width or x1 < 0 or y1 >= height or y1 < 0:
            game_close = True
        x1 += x1_change
        y1 += y1_change
        display.fill(blue)
        pygame.draw.rect(display, green, [foodx, foody, snake_block, snake_block])
        snake_head = []
        snake_head.append(x1)
        snake_head.append(y1)
        snake_list.append(snake_head)
        if len(snake_list) > length_of_snake:
            del snake_list[0]
        for x in snake_list[:-1]:
            if x == snake_head:
                game_close = True
        our_snake(snake_block, snake_list)
        pygame.display.update()
        if x1 == foodx and y1 == foody:
            foodx = round(random.randrange(0, width - snake_block) / 10.0) * 10.0
            foody = round(random.randrange(0, height - snake_block) / 10.0) * 10.0
            length_of_snake += 1
        clock.tick(snake_speed)
    pygame.quit()
    quit()
game_loop()

Jeu de Serpent (C)
Avec le même interface graphique que celle de python,simple du jeu du serpent en C.

#include SDL2/SDL.h
#include SDL2/SDL_ttf.h
#include stdbool.h
#include stdlib.h
#include time.h
#include stdio.h

#define WIDTH 600
#define HEIGHT 400
#define SNAKE_BLOCK 10
#define SNAKE_SPEED 5

typedef struct {
    int x;
    int y;
} Point;

SDL_Color white = {255, 255, 255, 255};
SDL_Color yellow = {255, 255, 102, 255};
SDL_Color black = {0, 0, 0, 255};
SDL_Color red = {213, 50, 80, 255};
SDL_Color green = {0, 255, 0, 255};
SDL_Color blue = {50, 153, 213, 255};

void render_snake(SDL_Renderer* renderer, Point snake[], int length) {
    for (int i = 0; i < length; i++) {
        SDL_Rect rect = {snake[i].x, snake[i].y, SNAKE_BLOCK, SNAKE_BLOCK};
        SDL_SetRenderDrawColor(renderer, black.r, black.g, black.b, black.a);
        SDL_RenderFillRect(renderer, &rect);
    }
}

void render_score(SDL_Renderer* renderer, int score) {
    TTF_Font* font = TTF_OpenFont("arial.ttf", 20);
    if (!font) {
        printf("Erreur chargement police: %s\n", TTF_GetError());
        return;
    }
    
    char score_text[50];
    snprintf(score_text, sizeof(score_text), "Score: %d", score);
    
    SDL_Surface* surface = TTF_RenderText_Solid(font, score_text, white);
    SDL_Texture* texture = SDL_CreateTextureFromSurface(renderer, surface);
    
    SDL_Rect rect = {10, 10, surface->w, surface->h};
    SDL_RenderCopy(renderer, texture, NULL, &rect);
    
    SDL_FreeSurface(surface);
    SDL_DestroyTexture(texture);
    TTF_CloseFont(font);
}

void game_loop() {
    SDL_Init(SDL_INIT_VIDEO);
    TTF_Init();
    
    SDL_Window* window = SDL_CreateWindow("Snake Game", SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, WIDTH, HEIGHT, 0);
    SDL_Renderer* renderer = SDL_CreateRenderer(window, -1, SDL_RENDERER_ACCELERATED);
    
    bool game_over = false;
    bool game_close = false;
    
    int x1 = WIDTH / 2;
    int y1 = HEIGHT / 2;
    int x1_change = 0;
    int y1_change = 0;
    
    Point snake[1000]; // Taille maximale du serpent
    int length_of_snake = 1;
    int score = 0;
    
    srand(time(NULL));
    int foodx = (rand() % (WIDTH - SNAKE_BLOCK)) / 10 * 10;
    int foody = (rand() % (HEIGHT - SNAKE_BLOCK)) / 10 * 10;
    
    while (!game_over) {
        while (game_close) {
            SDL_SetRenderDrawColor(renderer, blue.r, blue.g, blue.b, blue.a);
            SDL_RenderClear(renderer);
            
            TTF_Font* font = TTF_OpenFont("bahnschrift.ttf", 25);
            if (!font) {
                printf("Erreur chargement police: %s\n", TTF_GetError());
                return;
            }
            
            char message[100];
            snprintf(message, sizeof(message), "Score: %d - You Lost! Press C-Play Again or Q-Quit", score);
            
            SDL_Surface* surface = TTF_RenderText_Solid(font, message, red);
            SDL_Texture* texture = SDL_CreateTextureFromSurface(renderer, surface);
            
            SDL_Rect rect = {WIDTH / 6, HEIGHT / 3, surface->w, surface->h};
            SDL_RenderCopy(renderer, texture, NULL, &rect);
            
            SDL_FreeSurface(surface);
            SDL_DestroyTexture(texture);
            TTF_CloseFont(font);
            
            SDL_RenderPresent(renderer);
            
            SDL_Event event;
            while (SDL_PollEvent(&event)) {
                if (event.type == SDL_QUIT) {
                    game_over = true;
                    game_close = false;
                }
                if (event.type == SDL_KEYDOWN) {
                    if (event.key.keysym.sym == SDLK_q) {
                        game_over = true;
                        game_close = false;
                    }
                    if (event.key.keysym.sym == SDLK_c) {
                        game_close = false;
                        // Réinitialiser le jeu
                        x1 = WIDTH / 2;
                        y1 = HEIGHT / 2;
                        x1_change = 0;
                        y1_change = 0;
                        length_of_snake = 1;
                        score = 0;
                        foodx = (rand() % (WIDTH - SNAKE_BLOCK)) / 10 * 10;
                        foody = (rand() % (HEIGHT - SNAKE_BLOCK)) / 10 * 10;
                    }
                }
            }
        }
        
        SDL_Event event;
        while (SDL_PollEvent(&event)) {
            if (event.type == SDL_QUIT) {
                game_over = true;
            }
            if (event.type == SDL_KEYDOWN) {
                switch (event.key.keysym.sym) {
                    case SDLK_LEFT:
                        x1_change = -SNAKE_BLOCK;
                        y1_change = 0;
                        break;
                    case SDLK_RIGHT:
                        x1_change = SNAKE_BLOCK;
                        y1_change = 0;
                        break;
                    case SDLK_UP:
                        y1_change = -SNAKE_BLOCK;
                        x1_change = 0;
                        break;
                    case SDLK_DOWN:
                        y1_change = SNAKE_BLOCK;
                        x1_change = 0;
                        break;
                }
            }
        }
        
        if (x1 >= WIDTH || x1 < 0 || y1 >= HEIGHT || y1 < 0) {
            game_close = true;
        }
        
        x1 += x1_change;
        y1 += y1_change;
        
        SDL_SetRenderDrawColor(renderer, blue.r, blue.g, blue.b, blue.a);
        SDL_RenderClear(renderer);
        
        SDL_Rect food = {foodx, foody, SNAKE_BLOCK, SNAKE_BLOCK};
        SDL_SetRenderDrawColor(renderer, green.r, green.g, green.b, green.a);
        SDL_RenderFillRect(renderer, &food);
        
        Point head = {x1, y1};
        for (int i = length_of_snake - 1; i > 0; i--) {
            snake[i] = snake[i-1];
        }
        snake[0] = head;
        
        for (int i = 1; i < length_of_snake; i++) {
            if (snake[i].x == head.x && snake[i].y == head.y) {
                game_close = true;
            }
        }
        
        render_snake(renderer, snake, length_of_snake);
        render_score(renderer, score);
        SDL_RenderPresent(renderer);
        
        if (x1 == foodx && y1 == foody) {
            foodx = (rand() % (WIDTH - SNAKE_BLOCK)) / 10 * 10;
            foody = (rand() % (HEIGHT - SNAKE_BLOCK)) / 10 * 10;
            length_of_snake++;
            score += 10; // Augmente le score de 10 à chaque nourriture mangée
        }
        
        SDL_Delay(1000 / SNAKE_SPEED);
    }
    
    SDL_DestroyRenderer(renderer);
    SDL_DestroyWindow(window);
    TTF_Quit();
    SDL_Quit();
}

int main(int argc, char* argv[]) {
    game_loop();
    return 0;
}
